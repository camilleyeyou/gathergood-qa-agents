```typescript
import { z } from 'zod';

export const createOrgSchema = z.object({
  name: z.string().min(2, 'Organization name must be at least 2 characters'),
});
export type CreateOrgPayload = z.infer<typeof createOrgSchema>;

export const inviteMemberSchema = z.object({
  email: z.string().email('Enter a valid email'),
  role: z.enum(['MANAGER', 'VOLUNTEER']),  // OWNER intentionally excluded per TEAM-02
});
export type InviteMemberPayload = z.infer<typeof inviteMemberSchema>;

export const venueSchema = z.object({
  name: z.string().min(1, 'Venue name is required'),
  address: z.string().min(1, 'Address is required'),
  city: z.string().min(1, 'City is required'),
  state: z.string().min(1, 'State is required'),
  postal_code: z.string().min(1, 'Postal code is required'),
  capacity: z.coerce.number().int().positive('Capacity must be a positive number'),
  accessibility_info: z.string().optional(),
  parking_notes: z.string().optional(),
});
export type VenuePayload = z.infer<typeof venueSchema>;
```

**3. Create `frontend/src/lib/api/organizations.ts`:**

```typescript
import { apiClient } from './client';
import type { Organization, OrganizationMember, Venue } from '@/types/organizations';
import type { CreateOrgPayload, InviteMemberPayload, VenuePayload } from '@/lib/schemas/organizations';

export async function listOrgs(): Promise<Organization[]> {
  const response = await apiClient('/organizations/');
  if (!response.ok) throw new Error('Failed to fetch organizations');
  return response.json();
}

export async function createOrg(data: CreateOrgPayload): Promise<Organization> {
  const response = await apiClient('/organizations/', { method: 'POST', json: data });
  if (!response.ok) {
    const err = await response.json();
    throw new Error(err.detail || 'An organization with this name already exists. Try a more specific name.');
  }
  return response.json();
}

export async function getOrg(orgId: string): Promise<Organization> {
  const response = await apiClient(`/organizations/${orgId}/`);
  if (!response.ok) throw new Error('Failed to fetch organization');
  return response.json();
}

export async function updateOrg(orgId: string, data: Partial<CreateOrgPayload>): Promise<Organization> {
  const response = await apiClient(`/organizations/${orgId}/`, { method: 'PATCH', json: data });
  if (!response.ok) throw new Error('Could not save changes. Please try again.');
  return response.json();
}

export async function listMembers(orgId: string): Promise<OrganizationMember[]> {
  const response = await apiClient(`/organizations/${orgId}/members/`);
  if (!response.ok) throw new Error('Failed to fetch team members');
  return response.json();
}

export async function inviteMember(orgId: string, data: InviteMemberPayload): Promise<OrganizationMember> {
  const response = await apiClient(`/organizations/${orgId}/members/`, { method: 'POST', json: data });
  if (!response.ok) {
    const err = await response.json();
    throw new Error(err.detail || 'Could not send invite. Check the email and try again.');
  }
  return response.json();
}

export async function removeMember(orgId: string, memberId: string): Promise<void> {
  const response = await apiClient(`/organizations/${orgId}/members/${memberId}/`, { method: 'DELETE' });
  if (!response.ok) throw new Error('Failed to remove member');
}

export async function listVenues(orgId: string): Promise<Venue[]> {
  const response = await apiClient(`/organizations/${orgId}/venues/`);
  if (!response.ok) throw new Error('Failed to fetch venues');
  return response.json();
}

export async function createVenue(orgId: string, data: VenuePayload): Promise<Venue> {
  const response = await apiClient(`/organizations/${orgId}/venues/`, { method: 'POST', json: data });
  if (!response.ok) throw new Error('Could not save venue. Please try again.');
  return response.json();
}

export async function updateVenue(orgId: string, venueId: string, data: Partial<VenuePayload>): Promise<Venue> {
  const response = await apiClient(`/organizations/${orgId}/venues/${venueId}/`, { method: 'PATCH', json: data });
  if (!response.ok) throw new Error('Could not save venue. Please try again.');
  return response.json();
}
```
