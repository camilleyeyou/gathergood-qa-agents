```python
from django.contrib.auth import get_user_model
from common.utils import generate_unique_slug
from .models import Organization, OrganizationMember, Venue

User = get_user_model()


def create_organization(user, name: str) -> Organization:
    slug = generate_unique_slug(name, Organization)
    org = Organization.objects.create(name=name, slug=slug)
    OrganizationMember.objects.create(user=user, organization=org, role='OWNER')
    return org


def invite_member(org: Organization, email: str, role: str, inviting_user_role: str) -> OrganizationMember:
    """Invite a member. MANAGER cannot assign OWNER role."""
    if role == 'OWNER':
        raise ValueError("Cannot assign OWNER role via invite.")
    if inviting_user_role == 'MANAGER' and role == 'OWNER':
        raise ValueError("MANAGER cannot assign OWNER role.")

    user = User.objects.get(email=email)
    member, created = OrganizationMember.objects.get_or_create(
        user=user, organization=org,
        defaults={'role': role}
    )
    if not created:
        raise ValueError("User is already a member of this organization.")
    return member


def remove_member(org: Organization, member_id: str) -> None:
    """Remove a member from the organization."""
    member = OrganizationMember.objects.get(id=member_id, organization=org)
    if member.role == 'OWNER':
        owner_count = OrganizationMember.objects.filter(
            organization=org, role='OWNER'
        ).count()
        if owner_count <= 1:
            raise ValueError("Cannot remove the last owner of an organization.")
    member.delete()
```

```python
from rest_framework.permissions import BasePermission
from common.permissions import get_member_role


class IsOrgOwnerOrManager(BasePermission):
    """OWNER or MANAGER of the organization (extracted from URL kwargs)."""

    def has_permission(self, request, view):
        org = view.get_organization()
        if org is None:
            return False
        role = get_member_role(request.user, org)
        return role in ('OWNER', 'MANAGER')


class IsOrgOwner(BasePermission):
    """OWNER only."""

    def has_permission(self, request, view):
        org = view.get_organization()
        if org is None:
            return False
        return get_member_role(request.user, org) == 'OWNER'


class IsOrgMember(BasePermission):
    """Any member of the organization."""

    def has_permission(self, request, view):
        org = view.get_organization()
        if org is None:
            return False
        return get_member_role(request.user, org) is not None
```

```python
from rest_framework import status, generics
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from django.shortcuts import get_object_or_404

from .models import Organization, OrganizationMember, Venue
from .serializers import (
    OrganizationSerializer, OrganizationMemberSerializer,
    InviteMemberSerializer, VenueSerializer,
)
from .services import create_organization, invite_member, remove_member
from .permissions import IsOrgOwnerOrManager, IsOrgOwner, IsOrgMember
from common.permissions import get_member_role


class OrganizationListCreateView(APIView):
    permission_classes = [IsAuthenticated]

    def get(self, request):
        """ORG-02: List orgs where user is a member."""
        org_ids = OrganizationMember.objects.filter(
            user=request.user
        ).values_list('organization_id', flat=True)
        orgs = Organization.objects.filter(id__in=org_ids)
        serializer = OrganizationSerializer(orgs, many=True)
        return Response(serializer.data)

    def post(self, request):
        """ORG-01: Create org, auto-assign OWNER."""
        serializer = OrganizationSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)
        org = create_organization(request.user, serializer.validated_data['name'])
        return Response(
            OrganizationSerializer(org).data,
            status=status.HTTP_201_CREATED,
        )


class OrganizationDetailView(APIView):
    permission_classes = [IsAuthenticated]

    def get_organization(self):
        return get_object_or_404(Organization, pk=self.kwargs['org_id'])

    def get(self, request, org_id):
        org = self.get_organization()
        # Any member can view
        if get_member_role(request.user, org) is None:
            return Response(status=status.HTTP_403_FORBIDDEN)
        serializer = OrganizationSerializer(org)
        return Response(serializer.data)

    def patch(self, request, org_id):
        """ORG-03: OWNER/MANAGER can update org."""
        org = self.get_organization()
        role = get_member_role(request.user, org)
        if role not in ('OWNER', 'MANAGER'):
            return Response(status=status.HTTP_403_FORBIDDEN)
        serializer = OrganizationSerializer(org, data=request.data, partial=True)
        serializer.is_valid(raise_exception=True)
        serializer.save()
        return Response(serializer.data)


class MemberListCreateView(APIView):
    permission_classes = [IsAuthenticated]

    def get_organization(self):
        return get_object_or_404(Organization, pk=self.kwargs['org_id'])

    def get(self, request, org_id):
        """TEAM-03: Any org member can list team members."""
        org = self.get_organization()
        if get_member_role(request.user, org) is None:
            return Response(status=status.HTTP_403_FORBIDDEN)
        members = OrganizationMember.objects.filter(organization=org).select_related('user')
        serializer = OrganizationMemberSerializer(members, many=True)
        return Response(serializer.data)

    def post(self, request, org_id):
        """TEAM-01: OWNER/MANAGER can invite. TEAM-02: MANAGER cannot assign OWNER."""
        org = self.get_organization()
        role = get_member_role(request.user, org)
        if role not in ('OWNER', 'MANAGER'):
            return Response(
                {'detail': 'Only OWNER or MANAGER can invite members.'},
                status=status.HTTP_403_FORBIDDEN,
            )
        serializer = InviteMemberSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)
        try:
            member = invite_member(
                org=org,
                email=serializer.validated_data['email'],
                role=serializer.validated_data['role'],
                inviting_user_role=role,
            )
            return Response(
                OrganizationMemberSerializer(member).data,
                status=status.HTTP_201_CREATED,
            )
        except ValueError as e:
            return Response({'detail': str(e)}, status=status.HTTP_400_BAD_REQUEST)
        except Exception:
            return Response(
                {'detail': 'User not found or could not be invited.'},
                status=status.HTTP_400_BAD_REQUEST,
            )


class MemberDeleteView(APIView):
    permission_classes = [IsAuthenticated]

    def get_organization(self):
        return get_object_or_404(Organization, pk=self.kwargs['org_id'])

    def delete(self, request, org_id, member_id):
        """TEAM-04: Only OWNER can remove a member."""
        org = self.get_organization()
        role = get_member_role(request.user, org)
        if role != 'OWNER':
            return Response(
                {'detail': 'Only OWNER can remove members.'},
                status=status.HTTP_403_FORBIDDEN,
            )
        try:
            remove_member(org, member_id)
            return Response(status=status.HTTP_204_NO_CONTENT)
        except OrganizationMember.DoesNotExist:
            return Response(
                {'detail': 'Member not found.'},
                status=status.HTTP_404_NOT_FOUND,
            )
        except ValueError as e:
            return Response({'detail': str(e)}, status=status.HTTP_400_BAD_REQUEST)


class VenueListCreateView(APIView):
    permission_classes = [IsAuthenticated]

    def get_organization(self):

    def get(self, request, org_id):
        """VENU-02: List org venues."""
        org = self.get_organization()
        if get_member_role(request.user, org) is None:
            return Response(status=status.HTTP_403_FORBIDDEN)
        venues = Venue.objects.filter(organization=org)
        serializer = VenueSerializer(venues, many=True)
        return Response(serializer.data)

    def post(self, request, org_id):
        """VENU-01: Create venue."""
        org = self.get_organization()
        if get_member_role(request.user, org) is None:
            return Response(status=status.HTTP_403_FORBIDDEN)
        serializer = VenueSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)
        serializer.save(organization=org)
        return Response(serializer.data, status=status.HTTP_201_CREATED)


class VenueDetailView(APIView):
    permission_classes = [IsAuthenticated]

    def get_organization(self):
        return get_object_or_404(Organization, pk=self.kwargs['org_id'])

    def patch(self, request, org_id, venue_id):
        """VENU-03: Update venue."""
        org = self.get_organization()
        if get_member_role(request.user, org) is None:
            return Response(status=status.HTTP_403_FORBIDDEN)

        serializer = VenueSerializer(venue, data=request.data, partial=True)
        serializer.is_valid(raise_exception=True)

        return Response(serializer.data)
```
