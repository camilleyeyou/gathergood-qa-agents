**Test functions — 15 total, one per persona/flow combination:**

Each test follows this pattern:
```python
@pytest.mark.persona_agent
def test_{persona_name}_{flow}(agent_backend, claude_client, base_url, run_id, record_persona_result):
    """'{Persona Name}' persona attempts {flow} flow."""
    setup = _setup_{flow}_flow()
    agent_backend.page.goto(setup["start_url"], timeout=60000)
    agent_backend.page.wait_for_load_state("networkidle")

    result = run_persona_scenario(
        backend=agent_backend,
        client=claude_client,
        persona={PERSONA_CONSTANT},
        flow_goal=FLOW_GOALS["{flow}"],
        optimal_steps=OPTIMAL_STEPS["{flow}"],
        max_iterations={15 for tech_savvy/impatient, 25 for low_literacy/non_native/casual},
    )
    result["flow"] = "{flow}"
    save_persona_result(result, run_id)
    record_persona_result(result)

    # Persona tests assert friction_score is calculated, not PASS/FAIL
    assert result["friction_score"] is not None, "Friction score must be calculated"
    assert 1 <= result["friction_score"] <= 10, f"Friction score {result['friction_score']} out of range"
```
