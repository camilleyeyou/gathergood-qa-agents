```python
def _create_user_client(email=None, password="TestPass123!"):
    """Register a new user and return (email, authenticated httpx.Client)."""
    if email is None:
        email = unique_email()
    client = httpx.Client(base_url=_settings.API_URL, timeout=30)
    reg = client.post("/auth/register/", json={
        "email": email, "password": password, "password_confirm": password,
        "first_name": "Team", "last_name": "Member",
    })
    assert reg.status_code == 201
    login = client.post("/auth/login/", json={"email": email, "password": password})
    client.close()
    tokens = login.json()
    return email, httpx.Client(
        base_url=_settings.API_URL,
        headers={"Authorization": f"Bearer {tokens['access']}"},
        timeout=30,
    )
```

- Assert `"permission"` in `r.json().get("detail", "").lower()` (verified message: "Only managers and owners can perform this action.")

`test_volunteer_cannot_invite_member` (TPERM-02):
- Use `volunteer_client` to POST `/organizations/{org['slug']}/members/invite/` with `{"email": unique_email(), "role": "VOLUNTEER"}`
- Assert status 403
- Assert `"permission"` in `r.json().get("detail", "").lower()` (verified message: "You do not have permission to invite members.")

`test_manager_cannot_assign_owner_role` (TPERM-03):
- Use `manager_client` to POST `/organizations/{org['slug']}/members/invite/` with `{"email": unique_email(), "role": "OWNER"}`
- Assert status 403
- Assert `"Managers cannot assign the Owner role"` in `r.text`

`test_manager_cannot_remove_member` (TPERM-04):
